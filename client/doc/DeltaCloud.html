def self.define_class(name)
  @defined_classes ||= []
  if @defined_classes.include?(name)
    self.module_eval("API::#{name}")
  else
    @defined_classes << name unless @defined_classes.include?(name)
    API.const_set(name, Class.new)
  end
end
