def format
  sprintf("%-10s | %-15s | %-6s | %15s",
    @obj.id[0,10],
    @obj.storage_volume.respond_to?('name') ? @obj.storage_volume.name[0, 15] : 'unknown',
    @obj.state ? @obj.state[0,10] : 'unknown',
    @obj.created ? @obj.created[0,19] : 'unknown'
  )
end
